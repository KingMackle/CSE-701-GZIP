255 void GZipFileBlock::decode_dynamic_codes()
256 {
257  uint16_t numberOfLiteralCodes = 257 + parent.file.consumeNextNBits(5);
258  uint16_t numberOfDistanceCodes = 1 + parent.file.consumeNextNBits(5);
259  uint16_t numberOfCodeLengths = 4 + parent.file.consumeNextNBits(4);
260 
261  // decode the length of the codes
262  vector<uint16_t> lengthOfCodeLengths(19);
263  for (uint8_t i{0}; i < numberOfCodeLengths; i++)
264  {
265  lengthOfCodeLengths[CODE_LENGTHS_OF_CODE_LENGTH[i]] = parent.file.consumeNextNBits(3);
266  }
267  length_codes = constructCodes(lengthOfCodeLengths);
268  HuffmanTable lengthCodesTable(length_codes, lengthOfCodeLengths);
269  length_code_map = HuffmanTableToMap(lengthCodesTable);
270 
271  //read the lengths of the literal and distance codes
272  vector<uint16_t> allLengths(numberOfLiteralCodes + numberOfDistanceCodes);
273 
275 
276  /*
277  0 - 15: Represent code lengths of 0 - 15
278  16: Copy the previous code length 3 - 6 times.
279  The next 2 bits indicate repeat length
280  (0 = 3, ... , 3 = 6)
281  Example: Codes 8, 16 (+2 bits 11),
282  16 (+2 bits 10) will expand to
283  12 code lengths of 8 (1 + 6 + 5)
284  17: Repeat a code length of 0 for 3 - 10 times.
285  (3 bits of length)
286  18: Repeat a code length of 0 for 11 - 138 times
287  (7 bits of length)
288  */
289  for (size_t i{0}; i < allLengths.size(); i++) {
290  uint16_t symbol = readSymbol(length_codes, length_code_map);
291  uint16_t numberOfRepeats{0}, codeToRepeat{0};
292  // add the symbol or read the bits for its length
293  if (symbol == 16) {
294  // read the number of repeated codes
295  numberOfRepeats = 3 + parent.file.consumeNextNBits(2);
296  codeToRepeat = allLengths[i - 1];
297  } else if (symbol == 17) {
298  // read the number of repeated codes
299  numberOfRepeats = 3 + parent.file.consumeNextNBits(3);
300  } else if (symbol == 18) {
301  // read the number of repeated codes
302  numberOfRepeats = 11 + parent.file.consumeNextNBits(7);
303  } else {
304  allLengths[i] = symbol;
305  }
306  // do the code repeation if we need to
307  if (numberOfRepeats > 0) {
308  // repeat the codes
309  for (int j{0}; j < numberOfRepeats; j++){
310  allLengths[j + i] = codeToRepeat;
311  }
312  // advance the offset of the vector after repeats
313  i += (numberOfRepeats - 1);
314  }
315  }
316  // copy the literal length out of the main length vector
317  vector<uint16_t> lengthOfLiteralLengths(allLengths.begin(), allLengths.begin() + numberOfLiteralCodes);
318  // decode the length of the literals
319  literal_codes = constructCodes(lengthOfLiteralLengths);
320  HuffmanTable literalCodesTable(literal_codes, lengthOfLiteralLengths);
321  // create the literal code-length map
322  literal_code_map = HuffmanTableToMap(literalCodesTable);
323 
324  // copy the distance length out of the main length vector
325  vector<uint16_t> lengthOfDistanceLengths(allLengths.begin() + numberOfLiteralCodes, allLengths.end());
326  // decode the length of the distances
327  distance_codes = constructCodes(lengthOfDistanceLengths);
328  HuffmanTable distanceCodesTable(distance_codes, lengthOfDistanceLengths);

331 }

339 vector<uint16_t> GZipFileBlock::constructCodes(vector<uint16_t> &codeLengths)
340 {
341  // because we are dealing with unsigned and the code will never be more than 32768. init the vector to the Unsigned short max
342  vector<uint16_t> codes(codeLengths.size(), USHRT_MAX);
343  set<uint16_t> unique_codeLengths;
344  for_each(codeLengths.begin(), codeLengths.end(), [&unique_codeLengths](auto item) { if (item > 0) unique_codeLengths.insert(item); });
345  uint16_t code{0}, previousCodeLength{0};

347  for (auto it = unique_codeLengths.begin(); it != unique_codeLengths.end(); it++) {
348  auto length = *it;
349  // keep increasing the code value every iteration fo the loop
350  code <<= (length - previousCodeLength);
351  previousCodeLength = length;
352  // repeat the codes for all the offsets that had the same length
353  for (size_t i{0}; i < codeLengths.size(); i++) {
